```c
#include <stdio.h>
#include <stdlib.h>

main()
{
  int *p;
  int size = 5;

  // malloc関数でint型の配列を作成する
  p = (int *)malloc(sizeof(int) * size);

  // すべての要素に値を設定する
  // for文で適当な値を代入する
  for (int i = 0; i < size; i++) {
    p[i] = i * 2;
  }

  // 結果を出力する
  for (int i = 0; i < size; i++) {
    printf("%d\n", p[i]);
  }

  // メモリを解放する
  free(p);
}
```

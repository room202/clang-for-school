```c
#include <stdio.h>
#include <stdlib.h>

main()
{
  int *p;
  int size = 5;

  // callocを使用して、int型の配列を作成する
  // calloc(確保する区画の個数, 確保する領域のバイト数)
  p = (int *)calloc(size, sizeof(int));

  // すべての要素が0で初期化されていることを確認する
  for(int i = 0; i < size; i++) {
    printf("%d\n", p[i]);
  }

  // メモリを解放する
  free(p);
}
```
